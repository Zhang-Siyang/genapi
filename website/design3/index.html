How It Works

Three simple steps to implement your API clients

1

Define Your API Interface

Create a Go interface with annotations in comments

// @BaseURL("https://api.example.com/v1")
type UserAPI interface {
    genapi.Interface
    
    // @GET("/users")
    ListUsers(ctx context.Context) ([]User, error)
}
2

Generate the Client Code

Run the genapi tool to generate implementations

$ go generate ./...
# or directly
$ go run github.com/lexcao/genapi/cmd/genapi -file api.go
3

Use the Generated Client

Instantiate and use with full type safety

client := genapi.New[api.UserAPI]()
users, err := client.ListUsers(context.Background())

Examples

See GenAPI in action with a complete example

package api

import (
    "context"
    "github.com/lexcao/genapi"
)

//go:generate go run github.com/lexcao/genapi/cmd/genapi -file $GOFILE

// @BaseURL("https://jsonplaceholder.typicode.com")
type TodoAPI interface {
    genapi.Interface

    // @GET("/todos")
    GetTodos(ctx context.Context) ([]Todo, error)

    // @GET("/todos/{id}")
    GetTodo(ctx context.Context, id string) (Todo, error)

    // @POST("/todos")
    CreateTodo(ctx context.Context, todo Todo) (Todo, error)

    // @PUT("/todos/{id}")
    UpdateTodo(ctx context.Context, id string, todo Todo) (Todo, error)

    // @DELETE("/todos/{id}")
    DeleteTodo(ctx context.Context, id string) error
}

type Todo struct {
    UserID    int    `json:"userId"`
    ID        int    `json:"id"`
    Title     string `json:"title"`
    Completed bool   `json:"completed"`
}
// CODE GENERATED BY genapi. DO NOT EDIT.
package api

import (
    "context"
    "github.com/lexcao/genapi"
)

type implTodoAPI struct {
    client genapi.HttpClient
}

// SetHttpClient implments genapi.Interface
func (i *implTodoAPI) SetHttpClient(client genapi.HttpClient) {
    i.client = client
}

func (i *implTodoAPI) GetTodos(ctx context.Context) ([]Todo, error) {
    resp, err := i.client.Do(&genapi.Request{
        Method:  "GET",
        Path:    "/todos",
        Context: ctx,
    })
    return genapi.HandleResponse[[]Todo](resp, err)
}

func (i *implTodoAPI) GetTodo(ctx context.Context, id string) (Todo, error) {
    resp, err := i.client.Do(&genapi.Request{
        Method:     "GET",
        Path:       "/todos/{id}",
        PathParams: map[string]string{"id": id},
        Context:    ctx,
    })
    return genapi.HandleResponse[Todo](resp, err)
}
package main

import (
    "context"
    "fmt"
    "log"
    
    "yourproject/api"
    "github.com/lexcao/genapi"
)

func main() {
    // Create the client instance
    client := genapi.New[api.TodoAPI]()
    
    // Use the client methods with full type safety
    todos, err := client.GetTodos(context.Background())
    if err != nil {
        log.Fatalf("Error: %v", err)
    }
    
    fmt.Printf("Found %d todos\n", len(todos))
    
    // Create a new todo
    newTodo := api.Todo{
        Title:     "Learn GenAPI",
        Completed: false,
    }
    
    created, err := client.CreateTodo(context.Background(), newTodo)
    if err != nil {
        log.Fatalf("Error: %v", err)
    }
    
    fmt.Printf("Created todo: ID=%d\n", created.ID)
}

Annotation Reference

Documentation for all supported annotations

@BaseURL

Sets the base URL for all API requests

// @BaseURL("https://api.example.com/v1")

HTTP Methods

Define HTTP method and path for endpoints

// @GET("/users")
// @POST("/users")
// @PUT("/users/{id}")
// @DELETE("/users/{id}")

Path Parameters

URL parameters are mapped to method arguments

// @GET("/users/{id}")
GetUser(ctx context.Context, id string) (User, error)

Request Body

Non-primitive types are serialized as JSON in the request body

// @POST("/users")
CreateUser(ctx context.Context, user User) (User, error)
